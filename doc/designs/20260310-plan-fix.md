**Fixed code**:
```python
def is_subdataset_initialized(subdataset_path: Path) -> bool:
    """Check if subdataset has git tree available with files.

    A subdataset is initialized if:
    - .git exists (file or directory)
    - git rev-parse --show-toplevel returns THIS path (not parent)
    - Working tree has files (not empty)

    Args:
        subdataset_path: Path to subdataset directory

    Returns:
        True if subdataset is initialized with files, False otherwise
    """
    if not subdataset_path.exists():
        return False

    git_path = subdataset_path / ".git"
    if not git_path.exists():
        return False

    # Check if this is its own repository (not parent)
    try:
        result = subprocess.run(
            ["git", "-C", str(subdataset_path), "rev-parse", "--show-toplevel"],
            capture_output=True,
            timeout=5,
            check=True,
            text=True,
        )

        # The git root MUST be this subdataset path, not a parent
        git_root = Path(result.stdout.strip()).resolve()
        subdataset_resolved = subdataset_path.resolve()

        if git_root != subdataset_resolved:
            logger.debug(f"Not own repo: {subdataset_path} (root={git_root})")
            return False

        # Verify working tree has files (not empty except .git)
        non_hidden_items = [
            item for item in subdataset_path.iterdir()
            if not item.name.startswith(".")
        ]

        if not non_hidden_items:
            logger.debug(f"No files: {subdataset_path}")
            return False

        logger.debug(f"Initialized: {subdataset_path}")
        return True

    except (subprocess.CalledProcessError, subprocess.TimeoutExpired, FileNotFoundError):
        return False
```

**Changes**:
1. Use `git rev-parse --show-toplevel` instead of `git status`
2. Verify the git root IS the subdataset path (not parent)
3. Check for non-hidden files in working tree
4. Add debug logging for diagnostics

### STEP 2: Update Unit Tests

```python
def test_empty_directory_in_parent_repo_returns_false(self, tmp_path):
    """Empty directory inside parent repo should return False.

    Regression test for false positive bug where git status
    succeeds by finding parent repository.
    """
    # Create parent repository
    parent_repo = tmp_path / "parent"
    parent_repo.mkdir()
    subprocess.run(["git", "init"], cwd=parent_repo, check=True)

    # Create empty subdirectory (NOT a submodule)
    empty_subdir = parent_repo / "subdir"
    empty_subdir.mkdir()

    # Should return False (not initialized)
    assert not is_subdataset_initialized(empty_subdir)


def test_initialized_submodule_with_files_returns_true(self, tmp_path):
    """Properly initialized submodule with files should return True."""
    # Create parent repository
    parent_repo = tmp_path / "parent"
    parent_repo.mkdir()
    subprocess.run(["git", "init"], cwd=parent_repo, check=True)

    # Create remote repository with files
    remote_repo = tmp_path / "remote"
    remote_repo.mkdir()
    subprocess.run(["git", "init"], cwd=remote_repo, check=True)
    (remote_repo / "file.txt").write_text("content")
    subprocess.run(["git", "add", "."], cwd=remote_repo, check=True)
    subprocess.run(["git", "commit", "-m", "init"], cwd=remote_repo, check=True)

    # Add as submodule
    subprocess.run(
        ["git", "submodule", "add", str(remote_repo), "submodule"],
        cwd=parent_repo,
        check=True,
    )

    submodule_path = parent_repo / "submodule"

    # Should return True (fully initialized with files)
    assert is_subdataset_initialized(submodule_path)
```
